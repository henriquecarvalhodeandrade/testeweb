-- Cria o banco de dados
CREATE DATABASE IF NOT EXISTS sga_db;
USE sga_db;

-- Tabela de Usuários (para Autenticação)
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL, -- Onde o hash bcrypt será salvo [cite: 1819]
    data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabela de Cursos (Cadastro)
CREATE TABLE cursos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome_curso VARCHAR(150) NOT NULL UNIQUE,
    carga_horaria INT NOT NULL
);

-- Tabela de Alunos (Cadastro com Relacionamento)
CREATE TABLE alunos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    matricula VARCHAR(20) UNIQUE NOT NULL,
    curso_id INT, -- Chave Estrangeira [cite: 189]
    data_nascimento DATE,
    ativo BOOLEAN DEFAULT TRUE,
    FOREIGN KEY (curso_id) REFERENCES cursos(id) ON DELETE SET NULL
);
